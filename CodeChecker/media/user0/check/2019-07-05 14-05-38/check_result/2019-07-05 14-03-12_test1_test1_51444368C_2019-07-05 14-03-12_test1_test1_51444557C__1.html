#include<iostream>
usingnamespacestd;
intmat[501][501],tran[501][501],n,m;
intmain(){
cin>>n>>m;
for(inti=1;i<=n;i++)
for(intj=1;j<=m;j++)
cin>>mat[i][j];

for(inti=1;i<=n;i++)
for(intj=1;j<=m;j++)
cin>>tran[i][j];
for(inti=1;i<=n;i++){
for(intj=1;j<=m;j++){
if(mat[i][j]!=tran[i][j]){
for(intk=1;k<=(m-i);k++){
if(mat[i+k][j-k]==tran[i][j]){
mat[i][j]+=mat[i+k][j-k];
mat[i+k][j-k]=mat[i][j]-mat[i+k][j-k];
mat[i][j]-=mat[i+k][j-k];
}
}
if(mat[i][j]!=tran[i][j]){
cout<<"no"<<endl;
return0;
}
}
}
}
cout<<"yes"<<endl;
return0;
}
